ncfile = netCDF4.Dataset(filename, 'r+') # or 'w'

# create a new dimension
ncfile.createDimension('newdim', 1)

# create a new variable
ncfile.createVariable('newvar', 'f8', ('newdim', ))

# save the changes
ncfile.sync()

# close the files
ncfile.close()
